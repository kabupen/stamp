from scipy.stats import t as student

def e_y_ci(x0):
    low = a_hat_0 + a_hat_1 * x0 - student.ppf(0.025, n-2) * np.sqrt( ( 1/n + (x0 - df["x1"].mean() )**2/n/sxx ) * ve ) 
    high = a_hat_0 + a_hat_1 * x0 + student.ppf(0.025, n-2) * np.sqrt( ( 1/n + (x0 - df["x1"].mean() )**2/n/sxx ) * ve )
    
    return low, high
